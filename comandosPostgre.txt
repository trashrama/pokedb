drop table if exists demo;
drop table if exists items;
drop table if exists key_items;
drop table if exists lugar;
drop table if exists pokemons cascade;

create table tipos(id serial, nome varchar(30));

CREATE TABLE ITEMS(id_item serial not NULL, 
                   nome_item varchar(150) not NULL PRIMARY key, 
                   categoria varchar(50),
                   efeito varchar(300));
CREATE TABLE KEY_ITEMS(id_item serial not NULL, 
                   nome_item varchar(150) not NULL PRIMARY key, 
                   efeito varchar(300));
CREATE TABLE POKEMONS(id serial not null primary key,
                   nome varchar(50) not null,
                   id_tipo1 int not null,
                   id_tipo2 int,
                   hp int not null, 
                   attack int not NULL,
                   defense int not null, 
                   sp_atk int not null, 
                   sp_def int not null, 
                   especie varchar(30) not null,
                   speed int not null,
                   hab varchar(50) not null,
                   altura float not null, 
                   peso float not null,
                   nivel int not null,
                   FOREIGN KEY (id_tipo1) REFERENCES tipos(id),
                    FOREIGN KEY (id_tipo2) REFERENCES tipos(id));

create table lugar(id_poke int,
                  jogo varchar(10),
                  lugar varchar(50),
                  id REFERENCES pokemons (id)
                  );
                  
create table evolucaoNormal(id_atual int not null,
                      id_evo int not null UNIQUE,
                      nivel varchar(20) not null,
                     -- como faço pra linkar o tipo com a tabela de items quando for uma pedra?);
					-- a ideia que tive foi criar duas tabelas evolucoes para as evolucoes normais de niveis e trocas pokemons
                      -- e as outras de pedra
                      FOREIGN key (id_atual) REFERENCES pokemons (id),
                      FOREIGN key (id_evo) REFERENCES pokemons (id)
                      );
                      
create table evolucaoPedras(id_atual int not null,
                      id_evo int not null UNIQUE,
                      item varchar(50) not null,
                     -- como faço pra linkar o tipo com a tabela de items quando for uma pedra?);
					-- a ideia que tive foi criar duas tabelas evolucoes para as evolucoes normais de niveis e trocas pokemons
                      -- e as outras de pedra
                      FOREIGN key (id_atual) REFERENCES pokemons (id),
                      FOREIGN key (id_evo) REFERENCES pokemons (id),
                      FOREIGN key (item) references items (nome_item)
                      );
